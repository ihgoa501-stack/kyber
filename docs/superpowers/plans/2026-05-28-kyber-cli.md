# Kyber CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the `kyber` CLI tool — an interactive Agent scaffolder that generates Rust agent projects with control-theoretic trust guarantees.

**Architecture:** A Rust CLI using `clap` for commands and `dialoguer` for interactive dialogue. The CLI runs 6 rounds of Q&A to capture user intent, translates answers into a control system configuration (`kyber.toml`), and generates a complete Rust agent project from embedded templates. Two agent architectures are supported: `react` and `deep-verify`.

**Tech Stack:** Rust, clap 4.x, dialoguer 0.11, serde + toml, tera 1.x, include_dir, colored.

    }

}

        match self {

// kyber/src/dialogue/rounds.rs
use dialoguer::{theme::ColorfulTheme, Confirm, Input, MultiSelect, Select};
use colored::Colorize;
use crate::config::types::*;

pub fn run_dialogue(name: &str, template: String) -> DialogueAnswers {
    println!("\n{}", "═══ 设计你的 Agent ═══".cyan().bold());
    println!("我会问你几个问题，帮你搭建一个可控可靠的 Agent。\n");

    // Round 1: Role
    let role = round_role();
    println!();

    // Round 2: Tools
    let tools = round_tools();
    println!();

    // Round 3: Risk level
    let risk_level = round_risk();
    println!();

    // Round 4: Confirmation preferences
    let confirm_actions = round_confirm(&risk_level);
    println!();

    // Round 5: Architecture
    let architecture = if Confirm::new()
        .with_prompt(format!("使用 {} 架构（默认）？还是改用深度验证架构？回答 N 切换", template))
        .default(true)
        .interact()
        .unwrap()
    {
        match template.as_str() {
            "deep-verify" => ArchitectureStyle::DeepVerify,
            _ => ArchitectureStyle::React,
        }
    } else {
        match template.as_str() {
            "react" => ArchitectureStyle::DeepVerify,
            _ => ArchitectureStyle::React,
        }
    };
    println!();

    let answers = DialogueAnswers {
        name: name.to_string(),
        role,
        tools,
        risk_level,
        confirm_actions,
        architecture,
    };

    // Round 6: Summary & confirmation
    print_summary(&answers);
    let confirmed = Confirm::new()
        .with_prompt("确认生成？")
        .default(true)
        .interact()
        .unwrap();

    if !confirmed {
        println!("已取消。");
        std::process::exit(0);
    }

    answers
}

fn round_role() -> String {
    let roles = vec![
        "软件工程师（写代码、改代码、查文档）",
        "桌面助手（操作系统、管理文件）",
        "浏览器助手（上网查信息、填表单）",
        "自定义",
    ];

    let selection = Select::new()
        .with_prompt("这个 Agent 的主要角色是什么？")
        .theme(&ColorfulTheme::default())
        .items(&roles)
        .default(0)
        .interact()
        .unwrap();

    match selection {
        0 => "软件工程师".into(),
        1 => "桌面助手".into(),
        2 => "浏览器助手".into(),
        3 => Input::<String>::new()
            .with_prompt("请输入角色")
            .interact_text()
            .unwrap(),
        _ => unreachable!(),
    }
}

fn round_tools() -> Vec<Tool> {
    let tool_items = vec![
        "终端命令",
        "文件系统",
        "浏览器",
        "桌面应用",
        "网络请求",
        "git 操作",
    ];

    let selections = MultiSelect::new()
        .with_prompt("允许 Agent 使用哪些工具？（空格选择，回车确认）")
        .theme(&ColorfulTheme::default())
        .items(&tool_items)
        .interact()
        .unwrap();

    let tools: Vec<Tool> = selections.iter().map(|i| match i {
        0 => Tool::Terminal,
        1 => Tool::Filesystem,
        2 => Tool::Browser,
        3 => Tool::Desktop,
        4 => Tool::Network,
        5 => Tool::Git,
        _ => unreachable!(),
    }).collect();

    if tools.is_empty() {
        println!("{} 至少需要选择一个工具。", "⚠".yellow());
        round_tools() // retry
    } else {
        tools
    }
}

fn round_risk() -> RiskLevel {
    let risks = vec!["低 —— 试试无所谓，错了无妨", "中 —— 有点麻烦", "高 —— 出错很严重"];
    
    let selection = Select::new()
        .with_prompt("Agent 犯错的后果严重吗？")
        .theme(&ColorfulTheme::default())
        .items(&risks)
        .default(1)
        .interact()
        .unwrap();

    match selection {
        0 => RiskLevel::Low,
        1 => RiskLevel::Medium,
        2 => RiskLevel::High,
        _ => unreachable!(),
    }
}

fn round_confirm(risk: &RiskLevel) -> Vec<ActionKind> {
    let actions = vec![
        "读文件",
        "写文件",
        "删除文件",
        "执行命令",
        "git 操作",
        "网络请求",
    ];

    // Preselect based on risk level
    let defaults: Vec<bool> = match risk {
        RiskLevel::Low => vec![false, false, true, false, false, false],
        RiskLevel::Medium => vec![false, true, true, true, false, false],
        RiskLevel::High => vec![true, true, true, true, true, true],
    };

    let selections = MultiSelect::new()
        .with_prompt("哪些操作需要你点头确认才能执行？")
        .theme(&ColorfulTheme::default())
        .items(&actions)
        .defaults(&defaults)
        .interact()
        .unwrap();

    selections.iter().map(|i| match i {
        0 => ActionKind::Read,
        1 => ActionKind::Write,
        2 => ActionKind::Delete,
        3 => ActionKind::Execute,
        4 => ActionKind::Git,
        5 => ActionKind::Network,
        _ => unreachable!(),
    }).collect()
}

fn print_summary(answers: &DialogueAnswers) {
    println!("\n{}", "═══ 你的 Agent 配置 ═══".cyan().bold());
    println!("名称：     {}", answers.name);
    println!("角色：     {}", answers.role);
    println!("风险等级： {}", match answers.risk_level {
        RiskLevel::Low => "低",
        RiskLevel::Medium => "中",
        RiskLevel::High => "高",
    });
    println!("工具：     {}", answers.tools.iter()
        .map(|t| format!("{:?}", t))
        .collect::<Vec<_>>()
        .join(", "));

    let (threshold, max_iter, breaker) = compute_safety_params(&answers.risk_level, &answers.architecture);
    println!("\n{}", "你会得到这些保障：".green());
    println!("  ✓ 置信度低于 {} 时停住问人", threshold);
    println!("  ✓ 连续 {} 自动熔断", breaker);
    println!("  ✓ 最多 {} 步迭代，不会死循环", max_iter);
    println!("  ✓ 每步都有审计日志\n");
}

pub fn compute_safety_params(risk: &RiskLevel, arch: &ArchitectureStyle) -> (f64, u32, &'static str) {
    let base = match risk {
        RiskLevel::Low => (0.4, 50, "60 秒内 5 次失败"),
        RiskLevel::Medium => (0.6, 25, "60 秒内 3 次失败"),
        RiskLevel::High => (0.8, 15, "60 秒内 2 次失败"),
    };
    if matches!(arch, ArchitectureStyle::DeepVerify) {
        (base.0, base.1 * 2, base.2) // more iterations for deep-verify
    } else {
        base
    }
}

// kyber/src/dialogue/interpreter.rs
use crate::config::types::*;

impl From<DialogueAnswers> for KyberConfig {
    fn from(answers: DialogueAnswers) -> Self {
        let (confidence_threshold, max_iterations, circuit_breaker) = 
            match answers.risk_level {
                RiskLevel::Low => (0.4, 50, "5 failures in 60s"),
                RiskLevel::Medium => (0.6, 25, "3 failures in 60s"),
                RiskLevel::High => (0.8, 15, "2 failures in 60s"),
            };

        let max_iterations = if matches!(answers.architecture, ArchitectureStyle::DeepVerify) {
            max_iterations * 2
        } else {
            max_iterations
        };

        let safety_strategy = match answers.risk_level {
            RiskLevel::High => "deep_verify",
            _ => "verify_then_act",
        };

        KyberConfig {
            agent: AgentConfig {
                name: answers.name.clone(),
                role: answers.role.clone(),
                description: answers.role,
            },
            control: ControlConfig {
                architecture: answers.architecture,
                observer: ObserverConfig {
                    confidence_signals: vec!["logprobs".into(), "tool_success_rate".into()],
                    confidence_threshold,
                },
                controller: ControllerConfig {
                    max_iterations,
                    strategy: safety_strategy.into(),
                },
                safety: SafetyConfig {
                    require_confirm: answers.confirm_actions.iter()
                        .map(|a| format!("{:?}", a).to_lowercase())
                        .collect(),
                    circuit_breaker: circuit_breaker.into(),
                    audit_log: true,
                },
            },
            tools: ToolsConfig {
                enabled: answers.tools.iter()
                    .map(|t| format!("{:?}", t).to_lowercase())
                    .collect(),
            },
        }
    }
}

    pub fn new() -> Result<Self> {
        let mut tera = Tera::default();

        // Walk embedded templates and add each .tera file
        load_templates_recursive(&TEMPLATES_DIR, "", &mut tera)?;

        Ok(TemplateEngine { tera })
    }

    pub fn render(&self, template_name: &str, context: &Context) -> Result<String> {
        let rendered = self.tera.render(template_name, context)?;
        Ok(rendered)
    }
}

fn load_templates_recursive(dir: &Dir, prefix: &str, tera: &mut Tera) -> Result<()> {
    for file in dir.files() {
        let path = file.path();
        if let Some(ext) = path.extension() {
            if ext == "tera" {
                let relative = path.strip_prefix("/").unwrap_or(path);
                let name = relative.to_string_lossy().to_string();
                let content = std::str::from_utf8(file.contents())?;
                tera.add_raw_template(&name, content)?;
            }
        }
    }
    for subdir in dir.dirs() {
        let subprefix = format!("{}", subdir.path().display());
        load_templates_recursive(subdir, &subprefix, tera)?;
    }
    Ok(())
}

use std::time::{Duration, Instant};

/// Safety layer — the trust backbone of this Agent.
/// Every decision passes through here before execution.
pub struct SafetyLayer {
    iteration_count: u32,
    max_iterations: u32,
    failure_count: u32,
    failure_window: Vec<Instant>,
    circuit_breaker_threshold: u32,
    circuit_breaker_window_secs: u64,
    circuit_open: bool,
    require_confirm: Vec<String>,
    audit_log: Vec<AuditEntry>,
}

#[derive(Debug, Clone)]
pub struct AuditEntry {
    pub step: u32,
    pub confidence: f64,
    pub decision: String,
    pub result: String,
    pub timestamp: std::time::SystemTime,
}

impl SafetyLayer {
    pub fn new(config: &SafetyConfig) -> Self {
        // Parse circuit breaker: "3 failures in 60s"
        let parts: Vec<&str> = config.circuit_breaker.splitn(4, ' ').collect();
        let threshold: u32 = parts.first().and_then(|s| s.parse().ok()).unwrap_or(3);
        let window: u64 = parts.get(2).and_then(|s| s.parse().ok()).unwrap_or(60);

        SafetyLayer {
            iteration_count: 0,
            max_iterations: config.max_iterations,
            failure_count: 0,
            failure_window: Vec::new(),
            circuit_breaker_threshold: threshold,
            circuit_breaker_window_secs: window,
            circuit_open: false,
            require_confirm: config.require_confirm.clone(),
            audit_log: Vec::new(),
        }
    }

    /// Check if the Agent should stop (iteration limit)
    pub fn should_terminate(&self) -> bool {
        self.iteration_count >= self.max_iterations || self.circuit_open
    }

    /// Advance step counter
    pub fn advance_step(&mut self) {
        self.iteration_count += 1;
    }

    /// Check if an action requires human confirmation
    pub fn needs_confirm(&self, action_kind: &str) -> bool {
        self.require_confirm.contains(&action_kind.to_string())
    }

    /// Record an execution result and check circuit breaker
    pub fn record_result(&mut self, action: &str, success: bool) -> bool {
        self.audit_log.push(AuditEntry {
            step: self.iteration_count,
            confidence: 0.0, // filled by observer
            decision: action.to_string(),
            result: if success { "ok".into() } else { "fail".into() },
            timestamp: std::time::SystemTime::now(),
        });

        if !success {
            self.failure_count += 1;
            self.failure_window.push(Instant::now());

            // Prune old failures outside the window
            let cutoff = Instant::now() - Duration::from_secs(self.circuit_breaker_window_secs);
            self.failure_window.retain(|t| *t > cutoff);

            if self.failure_window.len() as u32 >= self.circuit_breaker_threshold {
                self.circuit_open = true;
                return false; // circuit open
            }
        }
        true
    }

    /// Print audit summary
    pub fn print_report(&self) {
        println!("\n═══ Audit Report ═══");
        println!("Steps executed: {}", self.iteration_count);
        println!("Failures: {}", self.failure_count);
        println!("Circuit open: {}", self.circuit_open);
    }
}

        if !self.terminal {
            return Err("Terminal tool not enabled".into());
        }
        let output = Command::new(cmd)
            .args(args)
            .output()
            .map_err(|e| format!("Command failed: {}", e))?;

        if output.status.success() {
            Ok(String::from_utf8_lossy(&output.stdout).to_string())
        } else {
            Err(String::from_utf8_lossy(&output.stderr).to_string())
        }
    }

    pub fn read_file(&self, path: &str) -> Result<String, String> {
        if !self.filesystem {
            return Err("Filesystem tool not enabled".into());
        }
        std::fs::read_to_string(path)
            .map_err(|e| format!("Read failed: {}", e))
    }

    pub fn write_file(&self, path: &str, content: &str) -> Result<(), String> {
        if !self.filesystem {
            return Err("Filesystem tool not enabled".into());
        }
        if let Some(parent) = std::path::Path::new(path).parent() {
            std::fs::create_dir_all(parent)
                .map_err(|e| format!("Create dir failed: {}", e))?;
        }
        std::fs::write(path, content)
            .map_err(|e| format!("Write failed: {}", e))
    }

use std::time::{SystemTime, UNIX_EPOCH};

/// Observer — estimates the Agent's internal state from observable signals.
/// In this React implementation, confidence is derived from:
///   - Success rate of recent tool calls
///   - Number of iterations
///   - (In production: logprobs from LLM API)
#[derive(Debug)]
pub struct Observer {
    confidence_threshold: f64,
    recent_successes: Vec<bool>,
    window_size: usize,
}

#[derive(Debug, Clone)]
pub struct Observation {
    pub confidence: f64,
    pub iteration_saturation: f64,
    pub timestamp: u64,
}

impl Observer {
    pub fn new(confidence_threshold: f64) -> Self {
        Observer {
            confidence_threshold,
            recent_successes: Vec::new(),
            window_size: 10,
        }
    }

    /// Take a measurement of the current state.
    /// confidence: inferred from recent history
    pub fn observe(&self) -> Observation {
        let confidence = self.estimate_confidence();

        Observation {
            confidence,
            iteration_saturation: 0.0,
            timestamp: SystemTime::now()
                .duration_since(UNIX_EPOCH)
                .unwrap()
                .as_secs(),
        }
    }

    /// Record a tool call result
    pub fn record_result(&mut self, success: bool) {
        self.recent_successes.push(success);
        if self.recent_successes.len() > self.window_size {
            self.recent_successes.remove(0);
        }
    }

    /// Apply user guidance — boosts confidence
    pub fn apply_hint(&mut self, _hint: &str) {
        // In a production agent, this would update a confidence prior
        // For now: user providing guidance counts as a success signal
        self.recent_successes.push(true);
        if self.recent_successes.len() > self.window_size {
            self.recent_successes.remove(0);
        }
    }

    fn estimate_confidence(&self) -> f64 {
        if self.recent_successes.is_empty() {
            return 0.5; // neutral starting confidence
        }
        let successes = self.recent_successes.iter().filter(|&&s| s).count();
        successes as f64 / self.recent_successes.len() as f64
    }
}

// kyber/templates/deep-verify/src/main.rs.tera
use std::io::{self, Write};
use anyhow::Result;

mod control;
mod tools;

use control::safety::SafetyLayer;
use control::observer::Observer;
use control::controller::Controller;
use control::verifier::Verifier;
use tools::ToolRegistry;

/// {{ agent_name }} — generated by Kyber (deep-verify architecture)
/// Trust guarantees:
///   - confidence gate: threshold = {{ confidence_threshold }}
///   - pre-execution verification (simulation sandbox)
///   - post-execution verification
///   - iteration limit: {{ max_iterations }} max steps
///   - circuit breaker: {{ circuit_breaker }}
///   - audit log: {{ audit_log }}
fn main() -> Result<()> {
    let config = load_config()?;

    let mut safety = SafetyLayer::new(&config.safety);
    let mut observer = Observer::new(config.observer.confidence_threshold);
    let mut controller = Controller::new(config.controller.max_iterations);
    let mut verifier = Verifier::new();
    let tools = ToolRegistry::new(true, true);

    println!("{} 已启动 (architecture: deep-verify)", "{{ agent_name }}");

    loop {
        safety.advance_step();

        // 1. Observe
        let observation = observer.observe();

        // 2. Confidence gate
        if observation.confidence < config.observer.confidence_threshold {
            println!("[step {}] 置信度偏低 ({:.2}), 请示用户", 
                     safety.iteration_count, observation.confidence);
            print!("请指导: ");
            io::stdout().flush()?;
            let mut hint = String::new();
            io::stdin().read_line(&mut hint)?;
            observer.apply_hint(hint.trim());
            continue;
        }

        // 3. Decide
        let action = controller.decide(&observation);
        println!("[step {}] 决策: {}", safety.iteration_count, action);

        // 4. Safety check
        if safety.needs_confirm(&action.kind()) {
            print!("执行 {}? [Y/n]: ", action);
            io::stdout().flush()?;
            let mut input = String::new();
            io::stdin().read_line(&mut input)?;
            if input.trim().to_lowercase() == "n" {
                println!("已取消");
                continue;
            }
        }

        // 5. Pre-verify (simulate before executing)
        match verifier.pre_verify(&action) {
            Ok(()) => println!("[step {}] 预验证通过", safety.iteration_count),
            Err(warning) => {
                println!("[step {}] 预验证警告: {}", safety.iteration_count, warning);
                print!("仍要执行？[Y/n]: ");
                io::stdout().flush()?;
                let mut input = String::new();
                io::stdin().read_line(&mut input)?;
                if input.trim().to_lowercase() == "n" {
                    continue;
                }
            }
        }

        // 6. Execute
        let result = action.execute(&tools);

        // 7. Post-verify
        let verification = verifier.post_verify(&action, &result);
        if let Err(issue) = verification {
            println!("[step {}] 后验证失败: {}", safety.iteration_count, issue);
            controller.handle_failure(&action);
            continue;
        }

        // 8. Record
        let ok = safety.record_result(&action, true);
        if !ok {
            println!("[step {}] 熔断触发！", safety.iteration_count);
            break;
        }

        // 9. Check termination
        if controller.is_done() || safety.should_terminate() {
            if safety.should_terminate() {
                println!("达到上限 ({} steps)，停止。", safety.max_iterations);
            } else {
                println!("任务完成。");
            }
            break;
        }
    }

    safety.print_report();
    Ok(())
}

// kyber/templates/deep-verify/src/observer.rs.tera
// Same structure as React observer, but with additional entropy tracking
// for the deep-verify architecture's extended confidence estimation

use std::time::{SystemTime, UNIX_EPOCH};

pub struct Observer {
    confidence_threshold: f64,
    recent_successes: Vec<bool>,
    window_size: usize,
    entropy_estimate: f64,
}

#[derive(Debug, Clone)]
pub struct Observation {
    pub confidence: f64,
    pub entropy: f64,
    pub iteration_saturation: f64,
    pub timestamp: u64,
}

impl Observer {
    pub fn new(confidence_threshold: f64) -> Self {
        Observer {
            confidence_threshold,
            recent_successes: Vec::new(),
            window_size: 15,
            entropy_estimate: 0.5,
        }
    }

    pub fn observe(&self) -> Observation {
        let confidence = self.estimate_confidence();
        Observation {
            confidence,
            entropy: self.entropy_estimate,
            iteration_saturation: 0.0,
            timestamp: SystemTime::now()
                .duration_since(UNIX_EPOCH)
                .unwrap()
                .as_secs(),
        }
    }

    pub fn record_result(&mut self, success: bool) {
        self.recent_successes.push(success);
        if self.recent_successes.len() > self.window_size {
            self.recent_successes.remove(0);
        }
        self.update_entropy();
    }

    pub fn apply_hint(&mut self, _hint: &str) {
        self.recent_successes.push(true);
        self.entropy_estimate *= 0.8;
        if self.recent_successes.len() > self.window_size {
            self.recent_successes.remove(0);
        }
    }

    fn estimate_confidence(&self) -> f64 {
        if self.recent_successes.is_empty() {
            return 0.5;
        }
        let successes = self.recent_successes.iter().filter(|&&s| s).count();
        (successes as f64 / self.recent_successes.len() as f64) * (1.0 - self.entropy_estimate)
    }

    fn update_entropy(&mut self) {
        if self.recent_successes.len() < 3 {
            self.entropy_estimate = 0.5;
            return;
        }
        let successes = self.recent_successes.iter().filter(|&&s| s).count() as f64;
        let len = self.recent_successes.len() as f64;
        let p = successes / len;
        if p <= 0.0 || p >= 1.0 {
            self.entropy_estimate = 0.1;
        } else {
            // Simplified entropy
            self.entropy_estimate = -(p * p.ln() + (1.0 - p) * (1.0 - p).ln()).min(1.0);
        }
    }
}

// kyber/templates/deep-verify/src/controller.rs.tera
use crate::tools::ToolRegistry;
use super::observer::Observation;

pub struct Controller {
    max_iterations: u32,
    steps_taken: u32,
    done: bool,
}

#[derive(Debug, Clone)]
pub struct Action {
    pub kind: String,
    pub description: String,
    pub params: std::collections::HashMap<String, String>,
}

impl Action {
    pub fn kind(&self) -> &str {
        &self.kind
    }

    pub fn execute(&self, tools: &ToolRegistry) -> Result<(), String> {
        match self.kind.as_str() {
            "read" => {
                let path = self.params.get("path").cloned()
                    .unwrap_or_else(|| "output.txt".into());
                tools.read_file(&path).map(|_| ())
            }
            "write" => {
                let path = self.params.get("path").cloned()
                    .unwrap_or_else(|| "output.txt".into());
                let content = self.params.get("content").cloned()
                    .unwrap_or_else(|| "".into());
                tools.write_file(&path, &content)
            }
            "execute" => {
                tools.execute_command("echo", &["deep-verify"]).map(|_| ())
            }
            "think" => Ok(()),
            "respond" => {
                println!("{}", self.description);
                Ok(())
            }
            _ => Err(format!("Unknown action: {}", self.kind)),
        }
    }
}

impl std::fmt::Display for Action {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        write!(f, "[{}] {} ({} params)", self.kind, self.description, self.params.len())
    }
}

impl Controller {
    pub fn new(max_iterations: u32) -> Self {
        Controller { max_iterations, steps_taken: 0, done: false }
    }

    pub fn decide(&mut self, obs: &Observation) -> Action {
        self.steps_taken += 1;
        if self.steps_taken >= self.max_iterations {
            self.done = true;
            return Action {
                kind: "respond".into(),
                description: "已完成所有步骤。".into(),
                params: std::collections::HashMap::new(),
            };
        }
        Action {
            kind: "think".into(),
            description: format!("deep-verify step {} (confidence: {:.2}, entropy: {:.2})",
                self.steps_taken, obs.confidence, obs.entropy),
            params: std::collections::HashMap::new(),
        }
    }

    pub fn is_done(&self) -> bool { self.done }
    pub fn handle_failure(&mut self, _action: &Action) {
        self.steps_taken = self.steps_taken.saturating_sub(1);
    }
}

// kyber/templates/deep-verify/src/verifier.rs.tera
use super::controller::Action;

/// Verifier — the extra trust layer in the deep-verify architecture.
/// Runs pre-execution and post-execution checks on every action.
pub struct Verifier {
    pre_checks_passed: u64,
    pre_checks_failed: u64,
    post_checks_passed: u64,
    post_checks_failed: u64,
}

impl Verifier {
    pub fn new() -> Self {
        Verifier {
            pre_checks_passed: 0,
            pre_checks_failed: 0,
            post_checks_passed: 0,
            post_checks_failed: 0,
        }
    }

    /// Pre-verify an action before execution.
    /// Returns Ok(()) if safe, Err(warning) if risky.
    pub fn pre_verify(&mut self, action: &Action) -> Result<(), String> {
        match action.kind() {
            "delete" => {
                // Extra scrutiny for deletions
                let path = action.params.get("path")
                    .map(|s| s.as_str())
                    .unwrap_or("unknown");
                if path == ".git" || path == "kyber.toml" || path == "Cargo.toml" {
                    self.pre_checks_failed += 1;
                    return Err(format!("尝试删除保护文件: {}", path));
                }
                self.pre_checks_passed += 1;
                Ok(())
            }
            "write" => {
                let path = action.params.get("path")
                    .map(|s| s.as_str())
                    .unwrap_or("unknown");
                if path.contains("..") || path.starts_with('/') {
                    self.pre_checks_failed += 1;
                    return Err(format!("路径可能不安全: {}", path));
                }
                self.pre_checks_passed += 1;
                Ok(())
            }
            "execute" => {
                let cmd = action.params.get("command")
                    .map(|s| s.as_str())
                    .unwrap_or("");
                if cmd.contains("rm -rf") || cmd.contains("sudo") {
                    self.pre_checks_failed += 1;
                    return Err(format!("高危命令: {}", cmd));
                }
                self.pre_checks_passed += 1;
                Ok(())
            }
            _ => {
                self.pre_checks_passed += 1;
                Ok(())
            }
        }
    }

    /// Post-verify an action after execution.
    /// Returns Ok(()) if result looks correct, Err if something is wrong.
    pub fn post_verify(&mut self, action: &Action, result: &Result<(), String>) -> Result<(), String> {
        match result {
            Ok(()) => {
                self.post_checks_passed += 1;
                Ok(())
            }
            Err(e) => {
                self.post_checks_failed += 1;
                Err(format!("执行失败: {}", e))
            }
        }
    }

    /// Print verifier statistics
    pub fn print_stats(&self) {
        println!("  Pre-verify:  {} passed, {} failed",
            self.pre_checks_passed, self.pre_checks_failed);
        println!("  Post-verify: {} passed, {} failed",
            self.post_checks_passed, self.post_checks_failed);
    }
}

// kyber/src/analysis/validate.rs
use colored::Colorize;
use crate::config::types::KyberConfig;

pub struct ValidationIssue {
    pub severity: Severity,
    pub message: String,
}

pub enum Severity {
    Error,
    Warning,
    Note,
}

pub fn validate_config(config: &KyberConfig) -> String {
    let mut issues = vec![];

    // 1. Consistency: risk ↔ confidence mapping
    //    (Can't reverse engineer risk from config, but we can check for contradictions)

    // 2. Confidence threshold bounds
    if config.control.observer.confidence_threshold < 0.1 {
        issues.push(ValidationIssue {
            severity: Severity::Warning,
            message: format!("置信度门限极低 ({:.1})，Agent 可能在不确定时也行动", 
                config.control.observer.confidence_threshold),
        });
    }
    if config.control.observer.confidence_threshold > 0.95 {
        issues.push(ValidationIssue {
            severity: Severity::Note,
            message: format!("置信度门限很高 ({:.1})，Agent 会频繁请示用户", 
                config.control.observer.confidence_threshold),
        });
    }

    // 3. Iteration bounds
    if config.control.controller.max_iterations == 0 {
        issues.push(ValidationIssue {
            severity: Severity::Error,
            message: "max_iterations 不能为 0".into(),
        });
    }
    if config.control.controller.max_iterations > 100 {
        issues.push(ValidationIssue {
            severity: Severity::Warning,
            message: format!("max_iterations {} 较大，注意死循环风险", 
                config.control.controller.max_iterations),
        });
    }
    if config.control.controller.max_iterations < 3 {
        issues.push(ValidationIssue {
            severity: Severity::Warning,
            message: "max_iterations 过小 (< 3)，Agent 可能无法完成任务".into(),
        });
    }

    // 4. Tools and confirm match
    let has_dangerous_tool = config.tools.enabled.iter().any(|t| {
        matches!(t.as_str(), "delete" | "execute" | "desktop")
    });
    let no_confirm = config.control.safety.require_confirm.is_empty();
    if has_dangerous_tool && no_confirm {
        issues.push(ValidationIssue {
            severity: Severity::Error,
            message: "启用了危险工具但没有设置确认要求，这是不安全的".into(),
        });
    }

    // 5. Audit log
    if !config.control.safety.audit_log {
        issues.push(ValidationIssue {
            severity: Severity::Warning,
            message: "审计日志已关闭，将无法追溯 Agent 行为".into(),
        });
    }

    format_report(&issues)
}

fn format_report(issues: &[ValidationIssue]) -> String {
    let mut report = String::new();
    report.push_str(&format!("\n{}", "═══ Kyber 安全检查报告 ═══".cyan().bold()));
    report.push('\n');

    let errors = issues.iter().filter(|i| matches!(i.severity, Severity::Error)).count();
    let warnings = issues.iter().filter(|i| matches!(i.severity, Severity::Warning)).count();

    if issues.is_empty() {
        report.push_str(&format!("{}\n", "✓ 所有检查通过，配置合理".green()));
    } else {
        for issue in issues {
            let prefix = match issue.severity {
                Severity::Error => "✗".red(),
                Severity::Warning => "⚠".yellow(),
                Severity::Note => "ℹ".blue(),
            };
            report.push_str(&format!("{} {}\n", prefix, issue.message));
        }
    }

    report.push('\n');
    report.push_str(&format!("{} 条错误，{} 条警告\n", errors, warnings));

    let grade = match (errors, warnings) {
        (0, 0) => "A（优秀）",
        (0, 1..=2) => "B（良好）",
        (0, _) => "C（一般）",
        _ => "F（不安全）",
    };
    report.push_str(&format!("安全等级：{}\n", grade));

    report
}

// kyber/src/analysis/stability.rs
use colored::Colorize;
use crate::config::types::{KyberConfig, RiskLevel, ArchitectureStyle};

pub struct StabilityMargin {
    pub score: f64,
    pub label: &'static str,
    pub description: String,
}

/// Estimate stability margin from configuration.
/// This is a heuristic, not a formal proof.
pub fn estimate_margin(config: &KyberConfig) -> String {
    // Infer risk level from confidence threshold
    let risk = if config.control.observer.confidence_threshold >= 0.7 {
        RiskLevel::High
    } else if config.control.observer.confidence_threshold >= 0.5 {
        RiskLevel::Medium
    } else {
        RiskLevel::Low
    };

    // "Gain" — how aggressively the Agent acts
    let gain = match risk {
        RiskLevel::Low => 0.9,
        RiskLevel::Medium => 0.6,
        RiskLevel::High => 0.3,
    };

    // "Damping" — how many safety layers slow it down
    let confirm_count = config.control.safety.require_confirm.len() as f64;
    let has_deep_verify = matches!(config.control.architecture, ArchitectureStyle::DeepVerify);
    let damping = 0.1 + confirm_count * 0.1 + if has_deep_verify { 0.3 } else { 0.0 };

    // Heuristic stability margin
    let margin = damping / (gain + 0.05);

    let label = if margin > 1.5 {
        "高稳定裕度"
    } else if margin > 0.8 {
        "中等稳定裕度"
    } else {
        "低稳定裕度"
    };

    let color_label = if margin > 1.5 {
        label.green()
    } else if margin > 0.8 {
        label.yellow()
    } else {
        label.red()
    };

    let mut report = String::new();
    report.push_str(&format!(
        "\n═══ 稳定裕度估算 ═══\n{}\n\n",
        color_label
    ));
    report.push_str(&format!("等效增益 (gain):      {:.2}\n", gain));
    report.push_str(&format!("等效阻尼 (damping):   {:.2}\n", damping));
    report.push_str(&format!("稳定裕度 (margin):    {:.2}\n", margin));
    report.push('\n');

    if margin < 0.8 {
        report.push_str("建议：增加确认操作或启用深度验证架构以提高稳定性\n");
    } else if margin > 2.0 {
        report.push_str("注意：稳定裕度很高，但 Agent 响应速度可能偏慢\n");
    } else {
        report.push_str("稳定性和响应速度的平衡合理。\n");
    }

    report
}
